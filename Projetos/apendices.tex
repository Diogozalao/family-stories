\chapter{Configuração de \emph{Deployment}}
\label{ap:deploy}

Resumo da topologia de \emph{deployment} (ver \Cref{sec:impl_deploy}):

\begin{itemize}[leftmargin=*]
  \item \textbf{Vercel} — \emph{frontend} (SPA), \emph{build} Vite,
  reescrita SPA.
  \item \textbf{Render} — \emph{backend} via Docker (FFmpeg, libmagic,
  Tesseract), \texttt{render.yaml}, sonda \texttt{/healthz},
  \texttt{CELERY\_ENABLED=False} e ChromaDB desativado.
  \item \textbf{Supabase} — Auth (JWKS), PostgreSQL (asyncpg), Storage.
\end{itemize}

\noindent\emph{Nota:} as variáveis sensíveis (chaves Supabase, Groq e
Gemini) são definidas no painel do Render e nunca versionadas no
repositório.

\begin{lstlisting}[language=Python,caption={Cliente LLM com cadeia de
\emph{fallback} de três níveis para texto (Groq $\rightarrow$ Ollama
$\rightarrow$ Gemini); o Gemini fica reservado para a visão
(\texttt{llm\_client.py}).},label={lst:llm}]
def generate(self, prompt: str, max_tokens: int = 1500) -> str:
    # Texto: Groq (70B, rápido) -> Ollama (offline/local) -> Gemini.
    # Groq primeiro porque o Ollama 3B local em CPU é demasiado lento.
    if self._groq_ok:                              # 1. Groq (nuvem, rápido)
        return self._groq_generate(prompt, max_tokens)
    if self._ollama_ok:                            # 2. local (offline/privado)
        return self._ollama_generate(prompt, max_tokens)
    return self._gemini_generate(prompt, max_tokens)  # 3. último recurso

def _groq_generate(self, prompt, max_tokens, _prev_error=None) -> str:
    try:
        resp = Groq(api_key=settings.GROQ_API_KEY).chat.completions.create(
            model=settings.GROQ_MODEL, messages=[{"role": "user",
            "content": prompt}], max_tokens=max_tokens, temperature=0.7)
        return resp.choices[0].message.content.strip()
    except Exception as groq_exc:                  # Groq falhou -> Gemini
        self._setup_gemini()
        return self._gemini_generate(prompt, max_tokens,
                                     _groq_error=str(groq_exc))
\end{lstlisting}

\begin{lstlisting}[language=Python,caption={Geração de cada fotograma do
efeito Ken Burns com \emph{easing smoothstep} e \emph{pan}
(\texttt{video\_builder.py}).},label={lst:kenburns}]
def make_frame(t: float) -> np.ndarray:
    progress = min(max(t / max(duration, 0.001), 0.0), 1.0)
    eased = progress * progress * (3.0 - 2.0 * progress)   # smoothstep
    zoom  = start_zoom + (end_zoom - start_zoom) * eased
    crop_w = int(TARGET_W / zoom); crop_h = int(TARGET_H / zoom)
    offset_x = int((eased - 0.5) * pan_dx)
    offset_y = int((eased - 0.5) * pan_dy)
    x = max(0, min(TARGET_W - crop_w, (TARGET_W - crop_w)//2 + offset_x))
    y = max(0, min(TARGET_H - crop_h, (TARGET_H - crop_h)//2 + offset_y))
    window = base[y:y+crop_h, x:x+crop_w]
    return np.array(Image.fromarray(window).resize((TARGET_W, TARGET_H)))
\end{lstlisting}

\begin{lstlisting}[language=Python,caption={Distribuição da narração de
cada cena pelas suas fotografias, com piso mínimo e compensação de
\emph{crossfade} (\texttt{video\_builder.py}).},label={lst:plandur}]
def plan_scene_durations(scene_audio_durations, scene_photo_counts):
    durations = []
    for audio_dur, n_photos in zip(scene_audio_durations, scene_photo_counts):
        if n_photos <= 0:
            durations.append(0.0); continue
        per_photo = max(MIN_SCENE_PHOTO_DURATION,
                        audio_dur / n_photos + CROSSFADE_SECONDS)
        durations.append(per_photo)
    return durations
\end{lstlisting}
